actor CoinItem : Inventory
{
	Tag "Money"
	+INVENTORY.KEEPDEPLETED
	+INVENTORY.UNDROPPABLE
	inventory.maxamount 1000000
}

actor SingleCoin : CustomInventory 25006
{
	//$Category Pickups (Wolf3D)
	//$Title Treasure (Coin, 1pts)
	//$Color 17
	inventory.pickupmessage "Picked up a coin."
	inventory.pickupsound "treasure/pickup"
	states
	{
	Spawn:
		COIN A -1 Light(COINLITE)
		stop
	Pickup:
		TNT1 A 0 A_GiveInventory("CoinItem")
		stop
	}
}

	Tag "Chest key"
	States
	{
	Spawn:
		SBKY A -1
		Stop
	}
}

ACTOR SupplyChest : SwitchingDecoration 11002
{
	//$Category Pickups (Wolf3D)
	//$Title Treasure Chest
	//$Color 17
	Height 34
	Radius 21
	+SOLID
	+USESPECIAL
	States
	{
	Spawn:
		SUBX A -1
		Stop
	Active:
		SUBX B 20
		SUBX C -1 A_UnSetSolid
		Stop
	}
