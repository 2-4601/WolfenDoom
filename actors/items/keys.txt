ACTOR BoABlueKey : Key replaces BlueCard
{
	//$Category Keys (Wolf3D)
	//$Title Blue Key (Blue Card)
	//$Color 13
	Inventory.PickupMessage "$GOTBLUEKEY"
	Inventory.Icon "STKEYS0"
	states
	{
	Spawn:
		BKEY A 10
		BKEY A 10 LIGHT(BOABKEY)
		Loop
	}
}
