ACTOR KTurretTracer : BulletTracer
{
	Radius 2
	Height 2
	Damage (random(10,30))
	Speed 200
	+CANNOTPUSH
	+DONTSPLASH
	+FORCEXYBILLBOARD
	+MISSILE
	+MTHRUSPECIES
	+NOBLOCKMAP
	+NODAMAGETHRUST
	+NOGRAVITY
	+NOTELEPORT
	+THRUSPECIES
	Alpha 0.75
	Scale 0.6
	RenderStyle "Add"
	Species "TURR"
	States
	{
	Spawn:
		PUFF A 0 NoDelay
		Goto Spawn2
	Spawn2:
		PUFF A 1 BRIGHT Light(3DCHAN)
		Loop
	Death:
	Crash:
		TNT1 AAA 0 A_SpawnItemEx("TracerSpark", 0, 0, 0, random(-2,2), random(-2,2), random(-2,2), random(0,359)) //T667 improvements
		TNT1 A 0 A_ChangeFlag("WINDTHRUST", 0)
		PUFF B 3 BRIGHT LIGHT(BPUFF1) A_PlaySound("ricochet")
		PUFF CD 3 BRIGHT LIGHT(BPUFF1)
		Stop
	XDeath:
		TNT1 A 0 A_ChangeFlag("WINDTHRUST", 0)
		TNT1 A 1 A_PlaySound("hitflesh")
		Stop
	}
}
