ACTOR Mine
{
	//$Category Hazards (Wolf3D)
	//$Title Mine
	//$Color 3
	Health 1000
	Radius 30
	Height 12
	Mass 9999999
	+ACTIVATEMCROSS
	+DONTGIB
	+LOOKALLAROUND
	+NOBLOOD
	+NOICEDEATH
	+NOTAUTOAIMED
	+SHOOTABLE
	+TOUCHY
	DeathSound "panzer/explode"
	Obituary "$OBMINE"
	States
	{
	Spawn:
		TNT1 A 0 NoDelay ACS_NamedExecuteAlways("SetRadarTID",0)
	See:
		MINE A 6 A_Chase
		Loop
	Death:
		"####" A 8 A_PlaySound("MINEF")
		"####" A 0 A_Scream
		"####" A 0 A_SpawnItemEx("GeneralExplosion_Medium")
		Stop
	}
}

ACTOR MineInvisible : Mine
{
	//$Title Mine (invisible)
	Radius 24
	Height 12
	RenderStyle "None"
}

ACTOR PlacedMine : Mine
{
	//$Title Friendly Mine (from Deployable item)
	Health 10
	Mass 500
	Obituary "$OBDMINE"
	States
	{
	Spawn:
		BOAM A 0 NoDelay A_CheckProximity("SpawnWait","BoaPlayer",128,8,CPXF_LESSOREQUAL | CPXF_SETTARGET | CPXF_CLOSEST)
	See:
		BOAM A 6 A_Chase
		Loop
	Death:
		"####" A 8 A_PlaySound("MINEF")
		"####" A 0 A_Scream
		"####" A 0 A_SpawnItemEx("FriendlyExplosion_Medium")
		Stop
	SpawnWait:
		"####" A 35
		Goto Spawn
	}
}

ACTOR UnderwaterMine : Mine
{
	//$Title Underwater Mine
	Health 300
	Radius 16
	Height 128
	+NOGRAVITY
	var int user_theta;
	States
	{
	Spawn:
	See:
		UWMR ABCDE 6 A_Chase
		Loop
	Death:
		TNT1 A 0 A_SpawnItemEx("GeneralExplosion_Medium",0,0,64)
		UWMR A 12 A_Scream
	Shock:
		UWMR A 0 A_SpawnProjectile("Bubble",random(64,76),0,user_theta,2)
		"####" A 0 A_SetUserVar("user_theta",user_theta+20)
		"####" A 0 A_JumpIf(user_theta==360,1)
		Loop
		UWMI H -1
		Stop
	}
}
