ACTOR ScientistUniform : SwitchableDecoration
{
	//$Category Pickups (Wolf3D)
	//$Title Scientist Uniform (useable)
	//$Color 13
	Radius 8
	Height 56
	-DORMANT
	+NOTAUTOAIMED
	+SOLID
	+USESPECIAL
	Activation THINGSPEC_Switch
	States
	{
	Spawn:
	Inactive:
		SCUN A -1 
		Stop
	Active:
		SCUN B 0 A_ChangeFlag("SOLID", FALSE)
		"####" B 0 A_SpawnItemEx("ScientistUniformPickup", 0, 0, 16, 0, 0, 0, 0, SXF_NOCHECKPOSITION)
		"####" B 0 A_ChangeFlag("DORMANT", FALSE)
		"####" B 0 A_ChangeFlag("USESPECIAL", FALSE)
		"####" B -1
		Stop
	}
}

ACTOR NullWeapon : Weapon
{
	Weapon.SelectionOrder 0
	Tag "no weapon (cloaked)"
	+WEAPON.NOALERT
	+INVENTORY.UNDROPPABLE
	States
	{
	Select:
		TNT1 A 0 SetPlayerProperty(0, TRUE, PROP_NOTARGET)
		TNT1 A 0 A_SetSpeed(0.85)
	Raise:
		TNT1 A 0 A_Raise
		Loop
	Deselect:
		TNT1 A 0 SetPlayerProperty(0, FALSE, PROP_NOTARGET)
		TNT1 A 0 A_SetSpeed(1.0)
		TNT1 A 0 {
				//A_TakeInventory("NullWeapon", 1); 	       //remove the weapon when selecting another weapon
				//A_TakeInventory("ScientistUniformToken", 1); //make the player look normal by removing the uniform token

			 }
	Lower:
		TNT1 A 0 A_Lower
		Loop
	Fire:
	Ready:
		TNT1 A 1 A_WeaponReady(WRF_NOFIRE)
		Loop
	}
}
