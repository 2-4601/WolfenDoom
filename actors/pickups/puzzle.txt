//Puzzle items here - Ozy81
ACTOR Rune_Tyr : PuzzleItem
{
	//$Category Puzzle Items (Wolf3D)
	//$Title Armanen Runes, Tyr
	//$Sprite RTYRA0
	//$Color 5
	PuzzleItem.Number 1
	PuzzleItem.FailMessage "$RUNES"
	Inventory.Amount 1
	Inventory.MaxAmount 1
	Inventory.Icon PTYRA0
	Inventory.PickupMessage "$RUNETYR"
	//Inventory.PickupSound "##"
	+COUNTITEM
	+NOGRAVITY
	+FLOATBOB
	+INVENTORY.INVBAR
	//ConversationID ### //use this if you plan to work with dialogues, and use different values for each actor
	Scale 0.2
	States
	{
	Spawn:
	TNT1 A 0 NoDelay A_PlaySound ("runes/loop", 7, frandom(0.8, 1.0), TRUE, ATTN_NORM)
	Floating:
	RTYR A -1
	Stop
	Pickup:
	TNT1 A 1 A_StopSound (7)
	Stop
	}
}
