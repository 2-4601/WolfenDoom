actor DecorationShip : SwitchableDecoration
{
	//$Category Props (Wolf3D)/Astrostein
	//$Title Space Ship (viper, activatable)
	//$Color 3
	SpawnID 211
	Radius 8
	Height 6
	Speed 6
	Damage 3
	SeeSound "viper/flyby"
	PROJECTILE
	+DONTSPLASH
	-ACTIVATEIMPACT
	+NoGravity
	+ACTIVATEPCROSS
	+CLIENTSIDEONLY
	States
	{
	Spawn:
	Active:
		POSS A 1
		POSS A 0 A_SpawnItemEx("DecorationShip_Steam", -32, -5, -1)
		POSS A 0 A_SpawnItemEx("DecorationShip_Steam", -32, 5, -1)
		Loop
	Inactive:
		POSS A -1
		Stop
	Death:
		POSS A 4
		Stop
	}
}
