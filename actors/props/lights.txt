ACTOR AlertLight
{
	+NOGRAVITY
	+INVISIBLE
	States
	{
		Spawn:
			UNKN A 1 NoDelay { if (args[0] == 0) { A_SetArg(0, 64); }}
		Static:
			UNKN A 1 A_LookEx(LOF_NOSOUNDCHECK | LOF_DONTCHASEGOAL | LOF_NOSEESOUND, 0, args[0] * 2.0, 0, 360, "Illuminate")
			Loop
		Illuminate:
			UNKN A 1 ACS_NamedExecuteAlways("AddExtraVisibility", 0, (args[0] * 2.0 - GetDistance(0)) / 1.0 + 8, 1)
			Goto Static
	}
}

ACTOR Chandelier1Switchable : SwitchableDecoration
{
	//$Category Light Sources (Wolf3D)
	//$Title Chandelier (1, switchable)
	//$Color 11
	
	//$Arg0 Sound
	//$Arg0Type 11
	//$Arg0Enum { 0 = "Yes"; 1 = "No"; }
	
	Radius 10
	Height 65 //same of sprite height
	+NOGRAVITY
	+SPAWNCEILING
	+SOLID
	
	States
	{
	Active:
		CHAN A 0 A_ChangeFlag("DORMANT", false) //mxd. DORMANT flag must be updated manually
	Spawn:
	  CHAN A 0 NoDelay A_PlaySound("FIRE_SMALL", CHAN_BODY, 0.6 - args[0], true)
	ActiveLoop:
		CHAN ABC 4 
		Loop
	Inactive:
		CHAN D 0 A_ChangeFlag("DORMANT", true) //mxd. DORMANT flag must be updated manually
		"####" D -1 A_StopSound(CHAN_BODY)
		Stop
	}
}
