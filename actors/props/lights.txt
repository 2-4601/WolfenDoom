ACTOR Chandelier1Switchable : SwitchableDecoration
{
	//$Category Light Sources (Wolf3D)
	//$Title Chandelier (1, switchable)
	//$Color 11
	
	//$Arg0 Sound
	//$Arg0Type 11
	//$Arg0Enum { 0 = "Yes"; 1 = "No"; }
	
	Radius 10
	Height 65 //same of sprite height
	Activation THINGSPEC_Switch
	+NOGRAVITY
	+SPAWNCEILING
	+SOLID
	States
	{
	Spawn:
		CHAN D -1
		Stop
	Active:
		CHAN A 4 A_PlaySound("DSCHAFIR", CHAN_BODY, 1.0 - args[0], 1)
		CHAN BC 4
		Loop
	Inactive:
		CHAN D -1 A_StopSound(CHAN_BODY)
		Stop
	}
}
