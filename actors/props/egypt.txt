ACTOR LavaCauldron
{
	//$Category Props (Wolf3D)
	//$Title Lava Cauldron (egypt)
	//$Color 3
	Radius 24
	Height 32
	+SOLID
	States
	{
	Spawn:
		LAVC A 0 NODELAY A_PlaySound("LavaCauldron/Loop",5,1,1,ATTN_STATIC)
		LAVC A 0 A_SpawnItemEx("LavaCauldronFire", 0, 0, Height, 0, 0, 0, 0, SXF_NOCHECKPOSITION)
	Bubble:
		LAVC ABC 6 Bright
		Loop
	}
}
