ACTOR HydrantShootable
{
	//$Category Props (Wolf3D)/Street
	//$Title Hydrant (shootable)
	//$Color 3
	Health 1
	Radius 10
	Height 42
	Mass 999999
	+SOLID
	+SHOOTABLE
	+NOBLOOD
	+DONTGIB
	+NOICEDEATH
	DeathSound "hydrant/death"
	States
	{
	Spawn:
		HYDR A -1
		Stop
	Death:
		HYDB A 4 A_Scream
		HYDB BC 4
		TNT1 A 0 A_PlaySound ("hydrant/broken", 7, frandom(0.8, 1.0), TRUE, ATTN_NORM)
	Broken:
		HYDR BCDE 4
		Loop
	}
}
