actor EnemyAlerter
{
	//$Category Misc (Wolf3D)
	//$Title Enemy Alerter (Radius 192)
	//$Color 1
	+LOOKALLAROUND
	Speed 0
	Meleerange 192
	//RenderStyle None
	States
	{
		Spawn:
			EXCL B 1 A_Look
			loop
		See:
			EXCL B 1 A_Chase
			loop
		Melee:
			EXCL A 1 A_AlertMonsters(512)
			Goto See
	}
}

//mxd
actor AlertMarker
{
	Scale 1.5
	
	+BRIGHT
	+NOGRAVITY
	+NOCLIP

	States
	{
		Spawn:
			EXCL C 1 NoDelay A_PlaySound("misc/alert", CHAN_BODY)
			goto Fade
		
		Fade:
			EXCL C 1 A_FadeOut(0.01)
			EXCL C 0 A_Warp(AAPTR_MASTER, 0, 0, 68)
			loop
	}
}
