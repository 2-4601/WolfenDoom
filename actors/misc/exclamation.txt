ACTOR Exclamation : SwitchableDecoration
{
	//$Category Misc (Wolf3D)
	//$Title Exclamation Mark
	//$Color 1
	Radius 32
	Height 64
	Scale 1.35
	RenderStyle "Add"
	+CLIENTSIDEONLY
	+NOGRAVITY
	States
	{
	Spawn:
		//TNT1 A -1 NODELAY ACS_NamedExecuteAlways("BoA_CompassQueue",0)
	Active:
		EXCL A -1 BRIGHT
		Loop
	Inactive:
		TNT1 A -1
		Stop
	}
}

ACTOR ExclamationTouchable : Exclamation
{
	//$Category Misc (Wolf3D)
	//$Title Exclamation Mark (inactive on touching)
	//$Color 1
	States
	{
	Spawn:
		//TNT1 A -1 NODELAY ACS_NamedExecuteAlways("BoA_CompassQueue",0)
	Active:
		EXCL A 1 NODELAY BRIGHT A_CheckProximity("Inactive", "BoAPlayer", 72, 1)
		Loop
	Inactive:
		EXCL A 35 BRIGHT
		EXCL B -1 BRIGHT
		Stop
	}	
}

ACTOR ExclamationInteraction : Exclamation //no need of GZDB descriptions
{
	Height 8
	Scale 1.0
	-SOLID
	+NOBLOCKMAP
	States
	{
	Spawn:
		TNT1 A 10 A_CheckProximity("Active", "BoAPlayer", 72, 1)
		Loop
	Active:
		EXCL A 1 BRIGHT A_CheckProximity("Inactive", "BoAPlayer", 72, 1)
		Loop
	Inactive:
		EXCL A 350
		Goto Spawn
	}	
}
